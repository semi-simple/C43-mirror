                    //PROGRAM TO DEMO DSZ AND GOTO. GET FIRST 4 PRIMES AFTER 1E100 TO STACK

-                          //SUBTRACT
"10" /                     //DIVIDE BY 10 TO GET SECONDS

"TIME "  //SEND TEXT TO C43

+
ALPHA                      //CHANGE TO ALPHA MODE
"ARROW UP AND DOWN TO START SHOW"

RETURN                     //STOPS EXECUTION
